            using (var db = new LiteDatabase(System.Web.Hosting.HostingEnvironment.MapPath("~/App_Data/OneData.db")))
            {

                var clientes = db.GetCollection<Cliente>("clientes");
                var results = clientes.FindAll();
                foreach (Cliente cli in results)
                {
                    lista.Add(cli);
                }
            }

            return View(lista);
